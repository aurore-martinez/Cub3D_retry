/*  Définitions de couleurs au format RGB(r,g,b) -> int 0xRRGGBB              */

#ifndef RGB_DEFS_H
# define RGB_DEFS_H

/* Macro de conversion (r,g,b) -> 0xRRGGBB */
# ifndef RGB
#  define RGB(r,g,b) (((r & 0xFF) << 16) | ((g & 0xFF) << 8) | (b & 0xFF))
# endif
